class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        
        lenth = len(nums)
        if(lenth == 0):
            return 0
        
        #----------------------
        num_sum = [0]*lenth
        num_sum[0] = nums[0]
        max_sum =  num_sum[0]
        
        for i in range(1,lenth):
            # include itself or itself and before
            num_sum[i] = max(nums[i], nums[i]+ num_sum[i-1])
            max_sum = max(max_sum, num_sum[i])
        
        return max_sum
